## üß† ‡¶â‡¶¶‡¶æ‡¶π‡¶∞‡¶£ (Examples)

```js

    {"id": 1, "x": 1},
    {"id": 2, "x": 9}

    {"id": 1, "x": 2, "y": 3},
```

```js
/**

 * @param {Array} arr2
 * @return {Array}

var join = function(arr1, arr2) {
  const map = {};

  // arr1 ‡¶•‡ßá‡¶ï‡ßá data ‡¶Ø‡ßã‡¶ó ‡¶ï‡¶∞‡¶æ
  for (const obj of arr1) {
    map[obj.id] = obj;

  // arr2 ‡¶•‡ßá‡¶ï‡ßá data merge ‡¶ï‡¶∞‡¶æ (overwrite ‡¶∏‡¶π)
  for (const obj of arr2) {
    if (map[obj.id]) {
      map[obj.id] = { ...map[obj.id], ...obj };
    } else {
      map[obj.id] = obj;
    }
  }

  // map ‚Üí array ‡¶§‡ßá ‡¶∞‡ßÇ‡¶™‡¶æ‡¶®‡ßç‡¶§‡¶∞ ‡¶è‡¶¨‡¶Ç sort ‡¶ï‡¶∞‡¶æ

};
```
